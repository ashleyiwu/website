    <!-- Firebase SDKs -->

    <script type="module">

        // Import the necessary Firebase modules

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, collection, onSnapshot, doc, setDoc, addDoc, updateDoc, deleteDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // Global variables provided by the Canvas environment

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

        const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}');

        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;


        // --- Firebase Initialization and Authentication ---

        let db, auth, userId;

        const userIdEl = document.getElementById('user-id');


        const initFirebase = async () => {

            try {

                const app = initializeApp(firebaseConfig);

                auth = getAuth(app);

                db = getFirestore(app);


                // Listen for authentication state changes

                onAuthStateChanged(auth, async (user) => {

                    if (user) {

                        userId = user.uid;

                        userIdEl.textContent = userId;

                        console.log("User authenticated:", userId);

                        // Once authenticated, load the tasks

                        loadTasks();

                    } else {

                        // User is signed out, handle as needed

                        userId = null;

                        userIdEl.textContent = 'Not authenticated';

                        console.log("User signed out.");

                    }

                });


                // Sign in using the custom token or anonymously if it's not available

                if (initialAuthToken) {

                    await signInWithCustomToken(auth, initialAuthToken);

                } else {

                    await signInAnonymously(auth);

                }


            } catch (error) {

                console.error("Error initializing Firebase or signing in:", error);

                userIdEl.textContent = 'Auth error';

            }

        };


        // --- DOM Elements ---

        const portfolioView = document.getElementById('portfolio-view');

        const tasksView = document.getElementById('tasks-view');

        const showPortfolioBtn = document.getElementById('show-portfolio');

        const showTasksBtn = document.getElementById('show-tasks');

        const titleEl = document.getElementById('title-text');

        const taskListEl = document.getElementById('task-list');

        const newTaskInput = document.getElementById('new-task-input');

        const addTaskBtn = document.getElementById('add-task-btn');


        // --- View Toggling Logic ---

        const showView = (viewId) => {

            portfolioView.classList.add('hidden');

            tasksView.classList.add('hidden');

            document.getElementById(viewId).classList.remove('hidden');

            titleEl.textContent = viewId === 'portfolio-view' ? 'My UX Portfolio' : 'My Tasks';

            

            // Highlight the active button

            showPortfolioBtn.classList.remove('bg-blue-600', 'hover:bg-blue-700', 'text-white');

            showPortfolioBtn.classList.add('bg-blue-100', 'hover:bg-blue-200', 'text-blue-800');

            showTasksBtn.classList.remove('bg-blue-600', 'hover:bg-blue-700', 'text-white');

            showTasksBtn.classList.add('bg-blue-100', 'hover:bg-blue-200', 'text-blue-800');

            

            if (viewId === 'portfolio-view') {

                showPortfolioBtn.classList.add('bg-blue-600', 'hover:bg-blue-700', 'text-white');

                showPortfolioBtn.classList.remove('bg-blue-100', 'hover:bg-blue-200', 'text-blue-800');

            } else {

                showTasksBtn.classList.add('bg-blue-600', 'hover:bg-blue-700', 'text-white');

                showTasksBtn.classList.remove('bg-blue-100', 'hover:bg-blue-200', 'text-blue-800');

            }

        };


        showPortfolioBtn.addEventListener('click', () => showView('portfolio-view'));

        showTasksBtn.addEventListener('click', () => showView('tasks-view'));


        // --- Task Manager Functions ---

        const getTasksCollection = () => {

            // Use the provided app ID and user ID to create a private collection path.

            // This ensures data is private to the user.

            const collectionPath = `/artifacts/${appId}/users/${userId}/tasks`;

            return collection(db, collectionPath);

        };


        const loadTasks = () => {

            if (!db || !userId) {

                console.warn("Firestore not ready or user not authenticated. Cannot load tasks.");

                return;

            }


            const tasksRef = getTasksCollection();

            // onSnapshot provides real-time updates for the task list

            onSnapshot(tasksRef, (snapshot) => {

                taskListEl.innerHTML = ''; // Clear the current list

                snapshot.docs.forEach(doc => {

                    const task = { id: doc.id, ...doc.data() };

                    renderTask(task);

                });

            }, (error) => {

                console.error("Error fetching tasks:", error);

            });

        };


        const renderTask = (task) => {

            const li = document.createElement('li');

            li.id = `task-${task.id}`;

            li.classList.add('flex', 'items-center', 'justify-between', 'p-4', 'rounded-lg', 'bg-white', 'border', 'border-gray-200', 'task-item');

            if (task.completed) {

                li.classList.add('completed');

            }


            const taskText = document.createElement('span');

            taskText.textContent = task.text;

            taskText.classList.add('flex-grow', 'text-gray-700', 'cursor-pointer');

            

            const actionContainer = document.createElement('div');

            actionContainer.classList.add('flex', 'space-x-2');


            // Complete button

            const completeBtn = document.createElement('button');

            completeBtn.innerHTML = `

                <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5 ${task.completed ? 'text-green-500' : 'text-gray-400'}" viewBox="0 0 20 20" fill="currentColor">

                    <path fill-rule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z" clip-rule="evenodd" />

                </svg>

            `;

            completeBtn.title = task.completed ? 'Mark as Incomplete' : 'Mark as Complete';

            completeBtn.classList.add('p-1', 'rounded-full', 'hover:bg-gray-100');

            completeBtn.addEventListener('click', () => toggleTaskCompletion(task));


            // Delete button

            const deleteBtn = document.createElement('button');

            deleteBtn.innerHTML = `

                <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5 text-gray-400 hover:text-red-500" viewBox="0 0 20 20" fill="currentColor">

                    <path fill-rule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm6 0a1 1 0 112 0v6a1 1 0 11-2 0V8z" clip-rule="evenodd" />

                </svg>

            `;

            deleteBtn.title = 'Delete Task';

            deleteBtn.classList.add('p-1', 'rounded-full', 'hover:bg-gray-100');

            deleteBtn.addEventListener('click', () => deleteTask(task.id));


            actionContainer.appendChild(completeBtn);

            actionContainer.appendChild(deleteBtn);


            li.appendChild(taskText);

            li.appendChild(actionContainer);

            taskListEl.appendChild(li);

        };

        

        const toggleTaskCompletion = async (task) => {

            if (!db) return;

            try {

                const taskRef = doc(db, getTasksCollection().path, task.id);

                await updateDoc(taskRef, { completed: !task.completed });

            } catch (error) {

                console.error("Error toggling task completion:", error);

            }

        };


        const deleteTask = async (taskId) => {

            if (!db) return;

            try {

                const taskRef = doc(db, getTasksCollection().path, taskId);

                await deleteDoc(taskRef);

            } catch (error) {

                console.error("Error deleting task:", error);

            }

        };


        const addTask = async () => {

            if (!db || !newTaskInput.value.trim()) return;

            try {

                await addDoc(getTasksCollection(), {

                    text: newTaskInput.value.trim(),

                    completed: false,

                    createdAt: new Date()

                });

                newTaskInput.value = ''; // Clear input

            } catch (error) {

                console.error("Error adding task:", error);

            }

        };


        addTaskBtn.addEventListener('click', addTask);

        newTaskInput.addEventListener('keydown', (e) => {

            if (e.key === 'Enter') {

                addTask();

            }

        });


        // Initialize everything when the window loads

        window.onload = () => {

            initFirebase();

            showView('portfolio-view'); // Show portfolio view by default

        };

    </script>
